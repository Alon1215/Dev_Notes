--- Mongo ---

(if on Ubuntu follow https://docs.mongodb.com/v3.2/tutorial/install-mongodb-on-ubuntu/)

	--- Install ---
	touch /etc/yum.repos.d/mongodb-org-3.2.repo
	sudo vim /etc/yum.repos.d/mongodb-org-3.2.repo

		[mongodb-org-3.2]
		name=MongoDB Repository
		baseurl=https://repo.mongodb.org/yum/amazon/2013.03/mongodb-org/3.2/x86_64/
		gpgcheck=1
		enabled=1
		gpgkey=https://www.mongodb.org/static/pgp/server-3.2.asc

	sudo yum install -y mongodb-org

	--- Start ---
	sudo service mongod start
	
	--- Start automatically on power on/reboot --- 
	systemctl enable mongod.service

	--- Stop ---
	sudo service mongod stop

	--- Restart ---
	sudo service mongod restart
